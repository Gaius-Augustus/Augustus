31 /* store cumulative sums of log likelihoods for each omega (optional: cumulative sum of number of substitutions)
32  * we create an object of cumValues for each bit_vector and reading frame combination
33  */
34 struct cumValues{
35  vector<double> logliks;
36  int numSubs;
37  int id;
38  cumValues(int i, int s=-1):numSubs(s), id(i){};
39  void addLogliks(vector<double>* ll){
40  if(logliks.size() == 0){
41  logliks.resize(ll->size(),0.0);
42  if(logliks.size() == 0){
43  cerr<<"logliks still empty!"<<endl;
44  }
45  }
46  for(int u = 0; u < ll->size(); u++){
47  logliks[u] += (*ll)[u];
48  }
49  }
50  void addNumSubs(int subs){
51  numSubs += subs;
52  }
53 };
