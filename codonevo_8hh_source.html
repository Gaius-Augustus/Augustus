33  // initialize pi with the uniform distribution on all sense codons

35  double normsum = 0.0;
36  for (unsigned c=0; c<64; c++){
37  if (GeneticCode::translate(c) == '*')
38  pi[c] = 0.0;
39  else
40  pi[c] = 1.0;
